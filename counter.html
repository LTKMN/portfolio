<body>


<button id="startButton">Start Timer</button>

<div id="display">$0.00</div>


<script>

    // === Configuration ===

    // Set the rate at which the money counts up (dollars per second)

    var rate = 0.05; // Change this value to set your desired rate


    // === Variables ===

    var timerStarted = false;

    var startTime;

    var intervalId;


    // === Event Listener for the Button ===

    document.getElementById('startButton').addEventListener('click', function() {

        if (!timerStarted) {

            timerStarted = true;

            startTime = new Date().getTime();

            intervalId = setInterval(updateDisplay, 100); // Update every 100 milliseconds

            this.textContent = 'Stop Timer';

        } else {

            timerStarted = false;

            clearInterval(intervalId);

            this.textContent = 'Start Timer';

        }

    });


    // === Function to Update the Display ===

    function updateDisplay() {

        var currentTime = new Date().getTime();

        var elapsedSeconds = (currentTime - startTime) / 1000;

        var moneyEarned = rate * elapsedSeconds;

        document.getElementById('display').textContent = '$' + moneyEarned.toFixed(2);

    }

</script>
